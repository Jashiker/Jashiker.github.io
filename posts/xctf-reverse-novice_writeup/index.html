import base64

def Redecode(cipher):
    message = ''
    cipher = base64.b64decode(cipher).decode()
    for i in cipher:
        x = ord(i) - 16
        x = x ^ 32
        message += chr(x)

    return message

cipher = 'XlNkVmtUI1MgXWBZXCFeKY+AaXNt'

flag = Redecode(cipher)
print(flag)
